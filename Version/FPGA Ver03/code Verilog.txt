module pwm_control_system (
    input wire clk,              // 100MHz
    input wire btnU,             // Nút tăng
    input wire btnD,             // Nút giảm
    input wire sw_failsafe,      // Công tắc bảo vệ (SW15)
    output reg pwm_out,
    output wire [3:0] led_level, // 4 LED hiển thị mức (0-10)
    output wire warning_led      // LED báo đang ở chế độ phá giới hạn
);

    parameter MAX_COUNT = 4000;
    parameter STEP      = 400;   // 10% của 4000
    parameter LIMIT_50  = 2000;  // Ngưỡng 50%

    reg [11:0] duty_reg = 0;     // Giá trị Duty Cycle mong muốn
    wire [11:0] final_duty;      // Giá trị Duty Cycle sau khi qua bộ Fail-safe
    reg [11:0] pwm_counter = 0;

    // --- MẠCH CHỐNG DỘI PHÍM (DEBOUNCE) ---
    // Giúp xác nhận việc nhấn nút chính xác, tránh nhảy số loạn xạ
    wire up_pressed, down_pressed;
    debounce db_up   (.clk(clk), .btn_in(btnU), .btn_out(up_pressed));
    debounce db_down (.clk(clk), .btn_in(btnD), .btn_out(down_pressed));

    // --- LOGIC ĐIỀU KHIỂN NÚT BẤM ---
    always @(posedge clk) begin
        if (up_pressed) begin
            if (duty_reg + STEP <= MAX_COUNT) 
                duty_reg <= duty_reg + STEP;
        end
        else if (down_pressed) begin
            if (duty_reg >= STEP) 
                duty_reg <= duty_reg - STEP;
        end
    end

    // --- CƠ CHẾ FAIL-SAFE (BẢO VỆ) ---
    // Theo đúng yêu cầu của bạn:
    // Nếu sw=0 và duty > 50% -> ép về 50%. Nếu sw=0 và duty <= 50% -> giữ nguyên.
    assign final_duty = (!sw_failsafe && duty_reg > LIMIT_50) ? LIMIT_50 : duty_reg;
    
    assign warning_led = sw_failsafe; // LED sáng báo hiệu đang tắt giới hạn
    assign led_level = duty_reg / 400; // Hiển thị mức từ 0-10 qua LED

    // --- TẠO XUNG PWM ---
    always @(posedge clk) begin
        pwm_counter <= (pwm_counter < MAX_COUNT - 1) ? pwm_counter + 1 : 0;
        pwm_out <= (pwm_counter < final_duty) ? 1'b1 : 1'b0;
    end

endmodule

// --- MODULE PHỤ: CHỐNG DỘI PHÍM ---
module debounce (
    input clk, input btn_in, output reg btn_out
);
    reg [19:0] count;
    reg btn_prev;
    always @(posedge clk) begin
        if (btn_in == btn_prev) begin
            if (count < 20'd1_000_000) count <= count + 1;
            else btn_out <= btn_in;
        end else begin
            count <= 0;
            btn_prev <= btn_in;
            btn_out <= 0;
        end
    end
endmodule

//Module phụ: chống dội phím
module debounce(
    input clk, 
    input btn_in,
    output reg btn_out
);

    reg [26:0] count; // dùng 26 thanh ghi để chứa giá trị 1 trăm triệu
    reg btn_pre; //trạng thái trước đó của nút
    
    always @(posedge clk) begin
        if(btn_in == btn_pre) begin // trường hợp 2 nút có giá trị bằng nhau --> đang ổn định
            if(count < 27'd100_000_000) count <= count + 1; //chưa đạt tới chu kỳ 100 triệu --> tiếp tục đếm thời gian
            else btn_out <= btn_in; //đã đủ chu kỳ 100 triệu --> xuất tín hiệu ra ngoài
        end else begin //trường hợp 2 nút có giá trị khác nhau (mới bấm hoặc đang bị nhiễu
            count <= 0; //reset bộ đếm về 0 để bắt đầu lại từ đầu
            btn_pre = btn_in;   // Cập nhật trạng thái mới để so sánh ở nhịp sau
            btn_out <= 0; // Trong lúc đang nhiễu thì tuyệt đối không xuất tín hiệu 1
        end
    end
 endmodule

//Phần module chính
module pwm_control(
    input wire clk,
    input wire btnUp,
    input wire btnDown,
    output reg pwm_out,
    output led_btnUp,
    output led_btnDown
);

assign led_btnUp = btnUp;
assign led_btnDown = btnDown;

parameter MAX_COUNT = 4000; //100.000.000 (100MHz) % 25.000 (kHz)
parameter STEP = 200; //change 5% each button

reg [11:0] duty_cycle_reg = 0;//giá trị Duty Cycle mong muốn, dùng 12 thanh ghi để chứa đủ giá trị 4000
reg [11:0] pwm_counter = 0; //tương tự như thanh ghi, để chứa đủ 4000 chu kỳ

//Mạch chống dội phím
wire up_pressed, down_pressed;
debounce db_up (.clk(clk), .btn_in(btnUp), .btn_out(up_pressed));
debounce dp_down (.clk(clk), .btn_in(btnDown), .btn_out(down_pressed));
// tác dụng: để lọc nhiễu
// ta dùng up_pressed và down_pressed đã được lọc nhiễu chứu không dùng btnUp và btnDOWN

//Logic điều khiển nút bấm
always @(posedge clk) begin
    if(up_pressed) begin //Kiểm tra xem tín hiệu từ Module Chống dội phím có đang ở mức Cao không.        if(duty_cycle_reg + STEP <= MAX_COUNT) // kiểm tra giá trị hiện tại của Duty Cycle có vượt quá 4000 ko --> ngăn chặn giá trị Duty Cycle vượt quá ngưỡng  --> VD vượt quá 150%
            duty_cycle_reg <= duty_cycle_reg + STEP;    
        end
        else if (down_pressed) begin //Nếu không tăng thì kiểm tra xem có lệnh giảm không
            if(duty_cycle_reg >= STEP) // ngặn chặn Duty Cycle giảm dưới ngưỡng --> VD dưới 0%
                duty_cycle_reg <= duty_cycle_reg - STEP;
        end
     end

// phần tạo xung PWM
always @(posedge clk) begin
    if(pwm_counter < MAX_COUNT - 1) begin
        pwm_counter = pwm_counter + 1; // nếu chưa chạm mốc thì tăng 1
    end else begin
        pwm_counter <= 0; //đã chạm mốc --> giảm thành 0   
    end
    
    if(pwm_counter < duty_cycle_reg) begin
        pwm_out = 1'b0; 
    end else begin
        pwm_out = 1'b1;
    end
end

endmodule
